                num_step_ways <- local({
                    memorization <- list()
                    function(n) {
                        memo_index <- as.character(n)
                        if (!is.null(memorization[[memo_index]])) return(memorization[[memo_index]])
                        if (n == 0) return(1)
                        if (n == 1) return(1)
                        if (n == 2) return(1)
                        ways <- num_step_ways(n - 1) + num_step_ways(n - 3)
                        memorization[[memo_index]] <<- ways 
                        return(ways)
                    }
                })
